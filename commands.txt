# the following is a list of all of the commands available in my startpage terminal
# click the back button to go back to the startpage

## GENERAL ##
[URL]
	Goes straight to that url, as long as it is valid.
[search query]
	Google's your query
ggl [search query]
	Google's your search query
duckduckgo | ddg
	Go to DuckDuckGo
help | halp
	Show this list

## KEYBOARD SHORTCUTS ##
[TAB]
	Resets the focus which enables you to use keyboard shorcuts for webpages
[W]
	Weather forecast for your location

## FIXES WHITE BACKGROUND ##
offline
	Offline mode, switches background to black
online
	Online mode, switches background to normal

## FUN / RANDOM ##
cout << [text]
	alert([text]);

## REDDIT ##
rdt
	Go to reddit
rdt [-r][-u][-s] [arg]
	-r  Go to subreddit [arg]
	-u  Go to user [arg]
	-s  Search [arg]
rdt [df][wg][up][sp]
	df  /r/Deliciousfails
	wg  /r/Weekendgunnit
	up  /r/Unixporn
	sp  /r/Startpages

## TWITTER ##
twt
	Go to twitter
twt [username]
	Go to specific twitter user @[username] ('@' is optional)
twt -s [text]
	Search twitter for tweets containing [text]
twt -su [username] [text]
	Search for tweets from @[username] containing [text] ('@' is optional)
twt -sh [tag]
	Search tweets containing #[tag] ('#' is optional)
twt -sh [username] [tag]
	Search tweets from @[username] containing #[tag] ('@' and '#' are optional)
twd
	Go to tweetdeck (in browser, obviously)

## INSTAGRAM ##
ig

ig [username]
